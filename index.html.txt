    <script>
        const tg = window.Telegram.WebApp;
        tg.expand();
        
        let map;
        let lat = 16.8661, lng = 96.1951; 

        function initMap() {
            map = new google.maps.Map(document.getElementById('map'), {
                center: { lat: lat, lng: lng },
                zoom: 14,
                disableDefaultUI: true
            });
        }

        function toggleService(type) {
            ['instant', 'hourly', 'daily', 'truck'].forEach(s => {
                document.getElementById(`form-${s}`).classList.add('hidden');
                document.getElementById(`tab-${s}`).className = "p-2 rounded-md text-gray-600";
            });
            document.getElementById(`form-${type}`).classList.remove('hidden');
            document.getElementById(`tab-${type}`).className = "bg-white p-2 rounded-md shadow-sm text-yellow-600 font-bold";
        }

        function switchRole() {
            const role = document.getElementById('userRole').value;
            if (role === 'driver') {
                document.getElementById('passenger-panel').classList.add('hidden');
                document.getElementById('driver-panel').classList.remove('hidden');
            } else {
                document.getElementById('passenger-panel').classList.remove('hidden');
                document.getElementById('driver-panel').classList.add('hidden');
            }
        }

        // TELEGRAM WEBAPP DATA SENDING LOGIC (SUITS BACKEND PERFECTLY)
        function submitKYC() {
            const name = document.getElementById('driver-name').value;
            const phone = document.getElementById('driver-phone').value;
            const carType = document.getElementById('driver-car-type').value;
            const carNumber = document.getElementById('driver-car-number').value;
            const township = document.getElementById('driver-township').value;

            if(!name || !phone || !carType || !carNumber || !township) {
                return tg.showAlert("အချက်အလက်များကို အပြည့်အစုံ ဖြည့်စွက်ပေးပါဗျာ။");
            }

            const kycData = { name, phone, carType, carNumber, township };
            
            // Bot ဆီသို့ စာသားစနစ်ဖြင့် တိုက်ရိုက်လှမ်းပို့ခြင်း
            tg.sendData("driver_register_kyc:" + JSON.stringify(kycData));
            tg.close();
        }

        function requestInstantRide() {
            const user = tg.initDataUnsafe?.user;
            const rideData = {
                passengerName: user?.first_name || "ခရီးသည်အမည်မသိ",
                passengerPhone: "App မှတစ်ဆင့်တောင်းဆိုသည်",
                pickup: "လက်ရှိတည်နေရာ (မြေပုံ)",
                dropoff: "မြို့တွင်းအမြန်သွားမည်"
            };
            tg.sendData("ride_request_instant:" + JSON.stringify(rideData));
            tg.close();
        }

        function requestHourlyRide() {
            const user = tg.initDataUnsafe?.user;
            const hours = document.getElementById('rental-hours').value;
            const rideData = {
                passengerName: user?.first_name || "ခရီးသည်အမည်မသိ",
                passengerPhone: "App မှတစ်ဆင့်တောင်းဆိုသည်",
                pickup: "လက်ရှိတည်နေရာ (မြေပုံ)",
                dropoff: "ရန်ကုန်မြို့တွင်း",
                hours: hours
            };
            tg.sendData("ride_request_hourly:" + JSON.stringify(rideData));
            tg.close();
        }

        function requestDailyRide() {
            const user = tg.initDataUnsafe?.user;
            const dest = document.getElementById('daily-destination').value;
            const date = document.getElementById('daily-date').value;
            const days = document.getElementById('daily-days').value;

            if(!dest || !date || !days) return tg.showAlert("အချက်အလက်အပြည့်အစုံ ဖြည့်ပါဗျာ။");

            const rideData = {
                passengerName: user?.first_name || "ခရီးသည်အမည်မသိ",
                passengerPhone: "App မှတစ်ဆင့်တောင်းဆိုသည်",
                pickup: "လက်ရှိတည်နေရာ",
                dropoff: dest,
                days: days
            };
            tg.sendData("ride_request_daily:" + JSON.stringify(rideData));
            tg.close();
        }

        function requestTruck() {
            const user = tg.initDataUnsafe?.user;
            const type = document.getElementById('truck-type').value;
            const from = document.getElementById('truck-from').value;
            const to = document.getElementById('truck-to').value;
            const details = document.getElementById('truck-details').value;

            if(!details) return tg.showAlert("အသေးစိတ်အချက်အလက်ကို ဖြည့်စွက်ပေးပါဗျာ။");

            const rideData = {
                passengerName: user?.first_name || "ခရီးသည်အမည်မသိ",
                passengerPhone: "App မှတစ်ဆင့်တောင်းဆိုသည်",
                pickup: from,
                dropoff: to,
                cargoType: type + " (" + details + ")"
            };
            tg.sendData("ride_request_truck:" + JSON.stringify(rideData));
            tg.close();
        }
